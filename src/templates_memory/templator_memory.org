# TREXIO in-memory (RAM) back end
#
# Same-process, disk-free back end: all data lives in the open handle. It
# mirrors the TEXT back end but with no file I/O: reading a group returns the
# in-RAM struct (an empty one if not yet present), and flushing is a no-op.
# The struct-based categories (numerical attributes, numerical/string
# datasets, string attributes) are stored in per-group structs; sparse,
# buffered and determinant data are stored in a generic name-keyed blob store
# on the handle. All data categories are therefore supported.

/* Generic in-RAM storage for the categories that the TEXT back end keeps in
   separate files: sparse datasets, buffered vectors and the determinant list.
   Each "blob" holds the growable arrays for one such dataset, looked up by
   name. This avoids adding per-dataset fields to the generated group structs. */
typedef struct trexio_mem_blob_s {
  char      name[256];
  int32_t*  index;     /* sparse: n_items * rank indices               */
  int64_t*  list;      /* determinant: n_items * rank bit-field ints   */
  double*   value;     /* sparse / buffered values (n_items)           */
  int64_t   n_items;
  int64_t   capacity;
  int32_t   rank;      /* indices per item (sparse) or ints/det (list) */
  struct trexio_mem_blob_s* next;
} trexio_mem_blob_t;

trexio_mem_blob_t*
trexio_mem_blob_find (trexio_memory_t* const file, const char* name)
{
  trexio_mem_blob_t* b = file->blobs;
  while (b != NULL) {
    if (strncmp(b->name, name, sizeof(b->name)) == 0) return b;
    b = b->next;
  }
  return NULL;
}

trexio_mem_blob_t*
trexio_mem_blob_get (trexio_memory_t* const file, const char* name, const int32_t rank)
{
  trexio_mem_blob_t* b = trexio_mem_blob_find(file, name);
  if (b != NULL) return b;
  b = (trexio_mem_blob_t*) calloc(1, sizeof(trexio_mem_blob_t));
  if (b == NULL) return NULL;
  strncpy(b->name, name, sizeof(b->name)-1);
  b->name[sizeof(b->name)-1] = '\0';
  b->rank = rank;
  b->next = file->blobs;
  file->blobs = b;
  return b;
}

void
trexio_mem_blob_free_all (trexio_memory_t* const file)
{
  trexio_mem_blob_t* b = file->blobs;
  while (b != NULL) {
    trexio_mem_blob_t* next = b->next;
    free(b->index);
    free(b->list);
    free(b->value);
    free(b);
    b = next;
  }
  file->blobs = NULL;
}

/* Grow the blob to hold at least `need` items. */
static trexio_exit_code
trexio_mem_blob_reserve (trexio_mem_blob_t* const b, const int64_t need,
                         const bool want_index, const bool want_list, const bool want_value)
{
  if (need <= b->capacity) return TREXIO_SUCCESS;
  int64_t cap = (b->capacity > 0) ? b->capacity : 1024L;
  while (cap < need) cap *= 2L;

  if (want_index) {
    int32_t* p = (int32_t*) realloc(b->index, (size_t) cap * (size_t) b->rank * sizeof(int32_t));
    if (p == NULL) return TREXIO_ALLOCATION_FAILED;
    b->index = p;
  }
  if (want_list) {
    int64_t* p = (int64_t*) realloc(b->list, (size_t) cap * (size_t) b->rank * sizeof(int64_t));
    if (p == NULL) return TREXIO_ALLOCATION_FAILED;
    b->list = p;
  }
  if (want_value) {
    double* p = (double*) realloc(b->value, (size_t) cap * sizeof(double));
    if (p == NULL) return TREXIO_ALLOCATION_FAILED;
    b->value = p;
  }
  b->capacity = cap;
  return TREXIO_SUCCESS;
}

trexio_exit_code
trexio_mem_blob_append_sparse (trexio_mem_blob_t* const b, const int64_t size,
                               const int32_t* index, const double* value)
{
  trexio_exit_code rc = trexio_mem_blob_reserve(b, b->n_items + size, true, false, true);
  if (rc != TREXIO_SUCCESS) return rc;
  memcpy(b->index + b->n_items * b->rank, index, (size_t) size * (size_t) b->rank * sizeof(int32_t));
  memcpy(b->value + b->n_items, value, (size_t) size * sizeof(double));
  b->n_items += size;
  return TREXIO_SUCCESS;
}

trexio_exit_code
trexio_mem_blob_read_sparse (const trexio_mem_blob_t* const b, const int64_t offset, const int64_t size,
                             int32_t* const index, double* const value, int64_t* const eof)
{
  if (b == NULL || offset >= b->n_items) { *eof = 0L; return TREXIO_END; }
  const int64_t avail = b->n_items - offset;
  const int64_t n = (size <= avail) ? size : avail;
  memcpy(index, b->index + offset * b->rank, (size_t) n * (size_t) b->rank * sizeof(int32_t));
  memcpy(value, b->value + offset, (size_t) n * sizeof(double));
  if (n < size) { *eof = n; return TREXIO_END; }
  return TREXIO_SUCCESS;
}
